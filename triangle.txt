#!/bin/bash
echo "Enter 3 sides of a triangle: "
read a
read b
read c
if [ $((a + $b)) -gt $c ] || [ $((a + c)) -gt $b ] || [ $((b + c)) -gt $a ]
then
        echo "These are sides of a triangle..."
else
        echo "Not sides of Triangle!"
fi
    if [ $a -eq $b ] && [ $b -eq $c ] && [ $a -eq $c ]
    then
        echo "Equilateral Triangle"
    elif [ $a -eq $b ] || [ $b -eq $c ] || [ $a -eq $c ]
    then
        echo "Isosceles Triangle"
    else
        echo "Scalene Triangle"
    fi
